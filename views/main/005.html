最长回文子串

给定一个字符串 s，找到 s 中最长的回文子串。你可以假设 s 的最大长度为 1000。

示例:

        
          输入: "babad"
          输出: "bab"
          注意: "aba" 也是一个有效答案。

          输入: "cbbd"
          输出: "bb"

解题：（M）

        
          var longestPalindrome = function(s) {
            function check(str) {
              return (
                str ===
                str
                  .split('')
                  .reverse()
                  .join('')
              );
            }
            var tmp = '';
            var maxStr = '';
            for (var i = 0; i < s.length; i++) {
              tmp = '';
              var char = s.charAt(i);
              for (let j = i; j < s.length; j++) {
                var char2 = s.charAt(j);
                tmp += char2;
                if (check(tmp)) {
                  maxStr = tmp.length > maxStr.length ? tmp : maxStr;
                }
              }
            }
            return maxStr;
          };
